1. 全局语言状态管理：

const { locale, t } = useI18n() const selectedLanguage = computed({ get: () => locale.value as LanguageCode, set: (value: LanguageCode) => { locale.value = value // 这会触发整个应用的语言切换 localStorage.setItem('yeyu_language', value) localStorage.setItem('user-locale', value) } })

2. 动态语言选项：

const languages = computed(() => { return supportedLanguages.map(lang => ({ code: lang.code, name: t(lang.key) || getLanguageNativeName(lang.code) })) })

3. 翻译文件结构：

// zh-CN.json { "settings": { "language": "语言", "languageChanged": "语言已切换" }, "languages": { "zhCN": "简体中文", "en": "英语", "ja": "日语" } } // en.json { "settings": { "language": "Language", "languageChanged": "Language changed" }, "languages": { "zhCN": "Simplified Chinese", "en": "English", "ja": "Japanese" } }
