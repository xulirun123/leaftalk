错误处理机制：

const handleLanguageClick = async () => { try { const route = router.resolve('/settings/language-selector') if (route.name === 'LanguageSelector') { await router.push('/settings/language-selector') } else { appStore.showToast('路由配置错误', 'error') } } catch (error) { console.error('跳转失败:', error) appStore.showToast('页面跳转失败', 'error') } }

setTimeout(async () => { try { router.back() } catch (error) { // 备选方案1：跳转到设置页面 try { await router.push('/settings/general') } catch (fallbackError) { // 备选方案2：跳转到首页 await router.push('/') } } }, 2000)
